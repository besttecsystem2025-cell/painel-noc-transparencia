import express from "express"; import axios from "axios"; const app = express(); const PORT = process.env.PORT || 3000; /* LISTA DOS PORTAIS */ const targets = [ { nome: "Cidade Ocidental", url: "https://transparencia.cidadeocidental.go.leg.br/" }, { nome: "Valparaíso", url: "https://transparencia.camaravalparparaiso.go.gov.br/" }, { nome: "São João d'Aliança", url: "https://transparencia.saojoaodalianca.go.leg.br/" }, { nome: "Alto Paraíso", url: "https://transparencia.altoparaisodegoias.go.leg.br/" }, { nome: "Luziânia", url: "https://transparencia.luziania.go.leg.br/" }, { nome: "Simolândia", url: "https://transparencia.simolandia.go.gov.br/" }, { nome: "Terezópolis", url: "https://transparencia.terezopolis.go.gov.br/" }, /* ACREÚNA CORRIGIDO */ { nome: "Acreúna", url: "https://transparencia.acreuna.go.gov.br/" } ]; /* MONITORAMENTO PROFUNDO */ async function checkSite(target) { const start = Date.now(); try { const response = await axios.get(target.url, { timeout: 10000, maxRedirects: 5, // permite redirecionamento correto validateStatus: () => true }); const tempo = Date.now() - start; let status = "ONLINE"; if (response.status >= 500) status = "OFFLINE"; else if (tempo > 800) status = "LENTO"; return { nome: target.nome, status, tempo, http: response.status }; } catch (error) { return { nome: target.nome, status: "OFFLINE", tempo: null, http: "erro" }; } } /* API */ app.get("/dados", async (req, res) => { const resultados = await Promise.all(targets.map(checkSite)); const online = resultados.filter(r => r.status === "ONLINE").length; const percentual = Math.round((online / resultados.length) * 100); res.json({ percentual, portais: resultados }); }); /* KEEP ALIVE */ setInterval(() => { if (process.env.RENDER_URL) { axios.get(process.env.RENDER_URL + "/dados").catch(() => {}); } }, 60000); app.listen(PORT, () => { console.log("Servidor rodando..."); });
